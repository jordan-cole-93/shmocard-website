**Server Action (`lib/waitlist.ts`):**
```typescript
'use server';
import { z } from 'zod'; // optional — can be hand-validated
const ALLOWED_PRODUCTS = new Set(['biz', 'link', 'reputation']);

export async function submitWaitlist(formData: FormData): Promise<{ ok: boolean; error?: string }> {
  // Honeypot
  if (formData.get('company_website')) return { ok: true };

  const email = String(formData.get('email') ?? '').trim();
  const product = String(formData.get('product') ?? '');

  // Server-side validation (REQ-08 spam protection minimum)
  if (!email || !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email)) return { ok: false, error: 'invalid-email' };
  if (!ALLOWED_PRODUCTS.has(product)) return { ok: false, error: 'invalid-product' };

  const url = process.env.GHL_WAITLIST_WEBHOOK_URL;
  if (!url) {
    console.warn('[waitlist] GHL_WAITLIST_WEBHOOK_URL not configured — capturing locally only');
    return { ok: true }; // graceful — UI shows success
  }

  try {
    const res = await fetch(url, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, product, submitted_at: new Date().toISOString() }),
    });
    if (!res.ok) return { ok: false, error: 'webhook-failed' };
    return { ok: true };
  } catch {
    return { ok: false, error: 'webhook-unreachable' };
  }
}
```
