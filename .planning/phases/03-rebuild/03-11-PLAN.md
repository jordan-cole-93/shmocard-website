**Reference (RESEARCH.md Pattern 4):**
```typescript
// app/api/revalidate/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { revalidateTag } from 'next/cache';
import crypto from 'node:crypto';

export const runtime = 'nodejs'; // crypto module not available on edge

export async function POST(req: NextRequest) {
  const rawBody = await req.text();
  const hmacHeader = req.headers.get('x-shopify-hmac-sha256') ?? '';
  const topic = req.headers.get('x-shopify-topic') ?? '';

  const secret = process.env.SHOPIFY_REVALIDATION_SECRET;
  if (!secret) return NextResponse.json({ error: 'misconfigured' }, { status: 500 });

  const computed = crypto
    .createHmac('sha256', secret)
    .update(rawBody, 'utf8')
    .digest('base64');

  // Length-safe + timing-safe
  const a = Buffer.from(computed);
  const b = Buffer.from(hmacHeader);
  if (a.length !== b.length || !crypto.timingSafeEqual(a, b)) {
    return NextResponse.json({ error: 'invalid hmac' }, { status: 401 });
  }

  let payload: any;
  try { payload = JSON.parse(rawBody); } catch {
    return NextResponse.json({ error: 'invalid json' }, { status: 400 });
  }

  const handle: string | undefined = payload?.handle;
  if (topic.startsWith('products/')) {
    revalidateTag('product');
    if (handle) revalidateTag(`product:${handle}`);
    revalidateTag('collection:shmo-review');
  }
  return NextResponse.json({ revalidated: true });
}
```

**Anti-patterns to avoid (RESEARCH.md Pitfalls 5):**
- DO NOT call `req.json()` first — that consumes the body and breaks HMAC computation
- DO NOT use `===` for HMAC compare — use `crypto.timingSafeEqual`
- DO NOT skip the length check before timingSafeEqual (it throws on mismatched buffers in older Node)

**Topics handled:** `products/create`, `products/update`, `products/delete`. Other topics: 200 OK noop (don't 4xx — Shopify retries on 4xx and we don't want noise).

**Edge runtime note:** `'use server'` not applicable here — route handler runtime is Node.js (set explicitly: `export const runtime = 'nodejs';`).
