**Allowlist regex (Server Action):**
```typescript
function assertCheckoutUrl(url: string): string {
  const u = new URL(url); // throws on malformed
  if (u.protocol !== 'https:') throw new Error('checkout: non-https scheme');
  if (u.username || u.password) throw new Error('checkout: embedded credentials');
  const allowedDomain = process.env.SHOPIFY_STORE_DOMAIN ?? '';
  const isMyshopify = u.hostname.endsWith('.myshopify.com');
  const isStoreDomain = allowedDomain && (u.hostname === allowedDomain || u.hostname === `shop.${allowedDomain}`);
  if (!isMyshopify && !isStoreDomain) throw new Error(`checkout: host ${u.hostname} not allowed`);
  return url;
}
```
